16  package net.sf.uadetector.datastore;
17  
18  import java.net.URL;
19  
20  import net.sf.uadetector.datareader.DataReader;
21  import net.sf.uadetector.datareader.XmlDataReader;
22  
23  /**
24   * This is the simplest implementation of a {@link DataStore}. It initialize the store by reading the <em>UAS data</em>
25   * via the passed URL and store it only in the Java heap space.
26   * 
27   * <p>
28   * The given resource must have a valid XML format with UTF-8 charset that validates against specified schema under <a
29   * href="http://user-agent-string.info/rpc/uasxmldata.dtd">http://user-agent-string.info/rpc/uasxmldata.dtd</a>.
30   * 
31   * @author André Rouél
32   */
33  public final class SimpleXmlDataStore extends AbstractDataStore implements DataStore {
34  
35  	/**
36  	 * The default data reader to read in <em>UAS data</em> in XML format
37  	 */
38  	private static final DataReader DEFAULT_DATA_READER = new XmlDataReader();
39  
40  	/**
41  	 * Constructs an {@code SimpleXmlDataStore} by reading <em>UAS data</em> by the specified default URL
42  	 * {@link DataStore#DEFAULT_DATA_URL} (in XML format).
43  	 * 
44  	 * @param dataUrl
45  	 *            URL to <em>UAS data</em>
46  	 * @param versionUrl
47  	 *            URL to version information about the given <em>UAS data</em>
48  	 */
49  	public SimpleXmlDataStore(final URL dataUrl, final URL versionUrl) {
50  		super(DEFAULT_DATA_READER, dataUrl, versionUrl, DEFAULT_CHARSET);
51  	}
52  
53  }
