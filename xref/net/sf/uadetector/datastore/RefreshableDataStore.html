16  package net.sf.uadetector.datastore;
17  
18  import javax.annotation.Nonnull;
19  
20  /**
21   * Extends the interface with an update functionality for <em>UAS data</em> in the store.
22   * 
23   * @author André Rouél
24   */
25  public interface RefreshableDataStore extends DataStore {
26  
27  	/**
28  	 * Message for the log if the read content can not be processed correctly
29  	 */
30  	String MSG_FAULTY_CONTENT = "The read content is faulty and can not be processed correctly.";
31  
32  	/**
33  	 * Message for the log if the <em>UAS data</em> can not be read from the given URL
34  	 */
35  	String MSG_URL_NOT_READABLE = "The data can not be read from the specified URL: %s";
36  
37  	/**
38  	 * This method returns a data store which will be used during start up and can be used in emergency cases.
39  	 * <p>
40  	 * This data store will be used instantly during initialization to avoid long initializations times of an
41  	 * {@link net.sf.uadetector.UserAgentStringParser}, especially when reading data by a network connection.
42  	 */
43  	@Nonnull
44  	DataStore getFallback();
45  
46  	/**
47  	 * Returns the update operation of this data store which can be triggered within an executor service.
48  	 * 
49  	 * @return an update operation
50  	 */
51  	@Nonnull
52  	UpdateOperation getUpdateOperation();
53  
54  	/**
55  	 * Triggers the update of the {@code DataStore}. When this action is executed, the current data URL will be read in
56  	 * and the {@code DataReader} parses and builds a new {@code Data} instance. Finally, the currently set {@code Data}
57  	 * reference will be replaced by the new one.
58  	 */
59  	void refresh();
60  
61  }
