272 	/**
273 	 * This method try to find by the given family name a matching enum value. The family name must match against an
274 	 * operating system entry in UAS data file.
275 	 * 
276 	 * @param family
277 	 *            name of an operating system family
278 	 * @return the matching enum value or {@code OperatingSystemFamily#UNKNOWN}
279 	 * @throws net.sf.qualitycheck.exception.IllegalNullArgumentException
280 	 *             if the given argument is {@code null}
281 	 */
282 	public static OperatingSystemFamily evaluate(@Nonnull final String family) {
283 		Check.notNull(family, "family");
284 
285 		OperatingSystemFamily result = UNKNOWN;
286 
287 		// search by name
288 		result = evaluateByName(family);
289 
290 		// search by pattern
291 		if (result == UNKNOWN) {
292 			result = evaluateByPattern(family);
293 		}
294 
295 		return result;
296 	}
297 
298 	/**
299 	 * This method try to find by the given family name a matching enum value. The family name will be evaluated against
300 	 * the stored name of an operating system entry.
301 	 * 
302 	 * @param family
303 	 *            name of an operating system family
304 	 * @return the matching enum value or {@code OperatingSystemFamily#UNKNOWN}
305 	 * @throws net.sf.qualitycheck.exception.IllegalNullArgumentException
306 	 *             if the given argument is {@code null}
307 	 */
308 	protected static OperatingSystemFamily evaluateByName(@Nonnull final String family) {
309 		Check.notNull(family, "family");
310 
311 		OperatingSystemFamily result = UNKNOWN;
312 		for (final OperatingSystemFamily value : values()) {
313 			if (value.getName().equals(family)) {
314 				result = value;
315 				break;
316 			}
317 		}
318 
319 		return result;
320 	}
321 
322 	/**
323 	 * This method try to find by the given family name a matching enum value. The family name will be evaluated against
324 	 * the stored regular expression of an operating system entry.
325 	 * 
326 	 * @param family
327 	 *            name of an operating system family
328 	 * @return the matching enum value or {@code OperatingSystemFamily#UNKNOWN}
329 	 * @throws net.sf.qualitycheck.exception.IllegalNullArgumentException
330 	 *             if the given argument is {@code null}
331 	 */
332 	protected static OperatingSystemFamily evaluateByPattern(@Nonnull final String family) {
333 		Check.notNull(family, "family");
334 
335 		OperatingSystemFamily result = UNKNOWN;
336 		for (final OperatingSystemFamily value : values()) {
337 			final Matcher m = value.getPattern().matcher(family);
338 			if (m.matches()) {
339 				result = value;
340 				break;
341 			}
342 		}
343 
344 		return result;
345 	}

347 	/**
348 	 * The internal family name in the UAS database.
349 	 */
350 	@Nonnull
351 	private final String name;
352 
353 	/**
354 	 * The regular expression which a family name must be match.
355 	 */
356 	@Nonnull
357 	private final Pattern pattern;
358 
359 	private OperatingSystemFamily(@Nonnull final String name, @Nonnull final Pattern pattern) {
360 		this.name = name;
361 		this.pattern = pattern;
362 	}
363 
364 	/**
365 	 * Gets the internal family name in the UAS database.
366 	 * 
367 	 * @return the internal family name
368 	 */
369 	@Nonnull
370 	public String getName() {
371 		return this.name;
372 	}
373 
374 	/**
375 	 * Gets the regular expression which a family name must be match with.
376 	 * 
377 	 * @return regular expression
378 	 */
379 	@Nonnull
380 	public Pattern getPattern() {
381 		return pattern;
382 	}
