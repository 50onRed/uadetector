16  package net.sf.uadetector;
17  
18  import javax.annotation.Nonnull;
19  
20  import net.sf.qualitycheck.Check;
21  import net.sf.uadetector.internal.data.domain.Robot;
22  
23  /**
24   * This enum represents the type of an user agent. The assignment to a type is performed within the UAS data.
25   * 
26   * @author André Rouél
27   */
28  public enum UserAgentType {
29  
30  	/**
31  	 * A web browser, especially on a desktop, notebook or workstation
32  	 */
33  	BROWSER("Browser"),
34  
35  	/**
36  	 * An email client, email reader
37  	 */
38  	EMAIL_CLIENT("Email client"),
39  
40  	/**
41  	 * A <b>news aggregator</b>, also termed a <b>feed aggregator<b>, <b>feed reader</b>, <b>news reader</b>, <b>RSS
42  	 * reader</b> or simply <b>aggregator</b>
43  	 */
44  	FEED_READER("Feed Reader"),
45  
46  	/**
47  	 * A library is a collection of resources used to develop software.
48  	 */
49  	LIBRARY("Library"),
50  
51  	/**
52  	 * Media player, also called multimedia player, is a term typically used to describe computer software for playing
53  	 * back multimedia files.
54  	 */
55  	MEDIAPLAYER("Multimedia Player"),
56  
57  	/**
58  	 * A mobile browser, also called a microbrowser, minibrowser, or wireless internet browser (WIB), is a web browser
59  	 * designed for use on a mobile device.
60  	 */
61  	MOBILE_BROWSER("Mobile Browser"),
62  
63  	/**
64  	 * Offline browser which may download completely or partially a website to a hard drive
65  	 */
66  	OFFLINE_BROWSER("Offline Browser"),
67  
68  	/**
69  	 * A software that doesn't match the other types
70  	 */
71  	OTHER("Other"),
72  
73  	/**
74  	 * Bots, such as Web crawlers
75  	 */
76  	ROBOT(Robot.TYPENAME),
77  
78  	/**
79  	 * An unknown user agent type
80  	 */
81  	UNKNOWN(""),
82  
83  	/**
84  	 * A software to hide the real user agent information
85  	 */
86  	USERAGENT_ANONYMIZER("Useragent Anonymizer"),
87  
88  	/**
89  	 * A software or service that validates parts of a website or webservice, such as CSS, HTML, JSON, XML
90  	 */
91  	VALIDATOR("Validator"),
92  
93  	/**
94  	 * A WAP browser is a web browser for mobile devices such as mobile phones that uses the Wireless Application
95  	 * Protocol (WAP). WAP is a technical standard for accessing information over a mobile wireless network.
96  	 */
97  	WAP_BROWSER("Wap Browser");
98  
99  	/**
100 	 * This method try to find by the given type name a matching enum value. The type name must match against an user
101 	 * agent entry in UAS data file.
102 	 * 
103 	 * @param typeName
104 	 *            name of an user agent type
105 	 * @return the matching enum value or {@code UserAgentType#UNKNOWN}
106 	 * @throws net.sf.qualitycheck.exception.IllegalNullArgumentException
107 	 *             if the given argument is {@code null}
108 	 */
109 	public static UserAgentType evaluateByTypeName(@Nonnull final String typeName) {
110 		Check.notNull(typeName, "typeName");
111 
112 		UserAgentType result = UNKNOWN;
113 		for (final UserAgentType value : values()) {
114 			if (value.getName().equals(typeName)) {
115 				result = value;
116 				break;
117 			}
118 		}
119 		return result;
120 	}
121 
122 	/**
123 	 * Name of the user agent type
124 	 */
125 	@Nonnull
126 	private final String name;
127 
128 	private UserAgentType(@Nonnull final String name) {
129 		this.name = name;
130 	}
131 
132 	/**
133 	 * Returns the name of the user agent type.
134 	 * 
135 	 * @return name of the type
136 	 */
137 	@Nonnull
138 	public String getName() {
139 		return name;
140 	}
141 
142 }
