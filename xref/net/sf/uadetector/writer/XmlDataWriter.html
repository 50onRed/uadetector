116 	private static Element createBrowser(final Browser browser, final Document doc) {
117 		final Element b = doc.createElement(Tag.BROWSER);
118 		final Element id = doc.createElement(Tag.ID);
119 		id.appendChild(doc.createTextNode(String.valueOf(browser.getId())));
120 		b.appendChild(id);
121 		final Element family = doc.createElement(Tag.TYPE);
122 		family.appendChild(doc.createTextNode(String.valueOf(browser.getType().getId())));
123 		b.appendChild(family);
124 		final Element name = doc.createElement(Tag.NAME);
125 		name.appendChild(doc.createTextNode(browser.getFamilyName()));
126 		b.appendChild(name);
127 		final Element url = doc.createElement(Tag.URL);
128 		url.appendChild(doc.createCDATASection(browser.getUrl()));
129 		b.appendChild(url);
130 		final Element company = doc.createElement(Tag.COMPANY);
131 		company.appendChild(doc.createCDATASection(browser.getProducer()));
132 		b.appendChild(company);
133 		final Element companyUrl = doc.createElement(Tag.URL_COMPANY);
134 		companyUrl.appendChild(doc.createCDATASection(browser.getProducerUrl()));
135 		b.appendChild(companyUrl);
136 		final Element icon = doc.createElement(Tag.ICON);
137 		icon.appendChild(doc.createTextNode(browser.getIcon()));
138 		b.appendChild(icon);
139 		final Element botInfoUrl = doc.createElement(Tag.BROWSER_INFO_URL);
140 		botInfoUrl.appendChild(doc.createTextNode(browser.getInfoUrl()));
141 		b.appendChild(botInfoUrl);
142 		return b;
143 	}

145 	private static Element createBrowserOperatingSystemMappings(final Data data, final Document doc) {
146 		final List<BrowserOperatingSystemMapping> mappings = new ArrayList<BrowserOperatingSystemMapping>(
147 				data.getBrowserToOperatingSystemMappings());
148 		Collections.sort(mappings, BrowserOperatingSystemMappingComparator.INSTANCE);
149 
150 		final Element browserTypesElement = doc.createElement(Tag.BROWSERS_OS);
151 		for (final BrowserOperatingSystemMapping mapping : mappings) {
152 			final Element t = doc.createElement(Tag.BROWSER_OS);
153 			final Element browserId = doc.createElement(Tag.BROWSER_ID);
154 			browserId.appendChild(doc.createTextNode(String.valueOf(mapping.getBrowserId())));
155 			t.appendChild(browserId);
156 			final Element osId = doc.createElement(Tag.OS_ID);
157 			osId.appendChild(doc.createTextNode(String.valueOf(mapping.getOperatingSystemId())));
158 			t.appendChild(osId);
159 			browserTypesElement.appendChild(t);
160 		}
161 		return browserTypesElement;
162 	}

164 	private static Element createBrowserPatterns(final Data data, final Document doc) {
165 		final List<BrowserPattern> patterns = new ArrayList<BrowserPattern>(data.getBrowserPatterns().size());
166 		for (final Entry<Integer, SortedSet<BrowserPattern>> entry : data.getBrowserPatterns().entrySet()) {
167 			patterns.addAll(entry.getValue());
168 		}
169 		Collections.sort(patterns, new OrderedPatternComparator<BrowserPattern>());
170 
171 		final Element browserTypesElement = doc.createElement(Tag.BROWSERS_REG);
172 		for (final BrowserPattern pattern : patterns) {
173 			final Element t = doc.createElement(Tag.BROWSER_REG);
174 			final Element order = doc.createElement(Tag.ORDER);
175 			order.appendChild(doc.createTextNode(String.valueOf(pattern.getPosition())));
176 			t.appendChild(order);
177 			final Element id = doc.createElement(Tag.BROWSER_ID);
178 			id.appendChild(doc.createTextNode(String.valueOf(pattern.getId())));
179 			t.appendChild(id);
180 			final Element family = doc.createElement(Tag.REGSTRING);
181 			family.appendChild(doc.createTextNode(RegularExpressionConverter.convertPatternToPerlRegex(pattern.getPattern())));
182 			t.appendChild(family);
183 			browserTypesElement.appendChild(t);
184 		}
185 		return browserTypesElement;
186 	}

215 	private static Element createDescription(@Nonnull final Data data, @Nonnull final Document doc) {
216 		final Element description = doc.createElement(Tag.DESCRIPTION);
217 		final Element label = doc.createElement(Tag.LABEL);
218 		description.appendChild(label).appendChild(
219 				doc.createTextNode("Data (format xml) for UASparser - http://user-agent-string.info/download/UASparser"));
220 		final Element version = doc.createElement("version");
221 		description.appendChild(version).appendChild(doc.createTextNode(data.getVersion()));
222 		final Element md5Checksum = doc.createElement("checksum");
223 		md5Checksum.setAttribute(Tag.TYPE, "MD5");
224 		description.appendChild(md5Checksum).appendChild(
225 				doc.createTextNode("http://user-agent-string.info/rpc/get_data.php?format=xml&md5=y"));
226 		final Element shaChecksum = doc.createElement("checksum");
227 		shaChecksum.setAttribute(Tag.TYPE, "SHA1");
228 		description.appendChild(shaChecksum).appendChild(
229 				doc.createTextNode("http://user-agent-string.info/rpc/get_data.php?format=xml&sha1=y"));
230 		return description;
231 	}

390 	static void transform(@Nonnull final Source xmlInput, @Nonnull final Result xmlOutput) throws TransformerException {
391 		Check.notNull(xmlInput, "xmlInput");
392 		Check.notNull(xmlOutput, "xmlOutput");
393 
394 		final TransformerFactory transformerFactory = TransformerFactory.newInstance();
395 		final Transformer transformer = transformerFactory.newTransformer();
396 		transformer.setOutputProperty(OutputKeys.INDENT, INDENT_OPTION);
397 		transformer.setOutputProperty(OutputKeys.DOCTYPE_SYSTEM, SCHEMA_URL);
398 		transformer.setOutputProperty("{http://xml.apache.org/xslt}indent-amount", INDENT_AMOUNT);
399 		transformer.transform(xmlInput, xmlOutput);
400 	}

402 	/**
403 	 * Transforms a given {@code Data} instance into XML and writes it to the passed in {@code OutputStream}.
404 	 * 
405 	 * @param data
406 	 *            {@code Data} to transform into XML
407 	 * @param outputStream
408 	 *            output stream to write
409 	 * @throws ParserConfigurationException
410 	 *             If a DocumentBuilder cannot be created which satisfies the configuration requested.
411 	 * @throws TransformerException
412 	 *             If an unrecoverable error occurs during the course of the transformation.
413 	 */
414 	public static void write(@Nonnull final Data data, @Nonnull final OutputStream outputStream) throws ParserConfigurationException,
415 			TransformerException {
416 		Check.notNull(data, "data");
417 		Check.notNull(outputStream, "outputStream");
418 
419 		final Document doc = newDocumentBuilder().newDocument();
420 
421 		// root element
422 		final Element uasdataElement = doc.createElement(Tag.UASDATA);
423 		doc.appendChild(uasdataElement);
424 
425 		// description element
426 		uasdataElement.appendChild(createDescription(data, doc));
427 
428 		// data element
429 		final Element dataElement = doc.createElement(Tag.DATA);
430 		uasdataElement.appendChild(dataElement);
431 
432 		dataElement.appendChild(createRobots(data, doc));
433 		dataElement.appendChild(createOperatingSystems(data, doc));
434 		dataElement.appendChild(createBrowsers(data, doc));
435 		dataElement.appendChild(createBrowserTypes(data, doc));
436 		dataElement.appendChild(createBrowserPatterns(data, doc));
437 		dataElement.appendChild(createBrowserOperatingSystemMappings(data, doc));
438 		dataElement.appendChild(createOperatingSystemPatterns(data, doc));
439 		dataElement.appendChild(createDevices(data, doc));
440 		dataElement.appendChild(createDevicePatterns(data, doc));
441 
442 		// write the content to output stream
443 		final DOMSource source = new DOMSource(doc);
444 		final StreamResult result = new StreamResult(outputStream);
445 		transform(source, result);
446 	}
