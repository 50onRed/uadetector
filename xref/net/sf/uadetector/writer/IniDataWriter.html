46  /**
47   * This utility is intended to transform an instance of {@code Data} into an <i>UAS data</i> conform XML document and
48   * allows us to recreate an <code>uas.xml</code>.
49   * 
50   * @author André Rouél
51   */
52  @ThreadSafe
53  public final class IniDataWriter {
54  
55  	interface Char {
56  		char EQUALS = '=';
57  		char NEWLINE = '\n';
58  		char QUOTE = '"';
59  		char SEMICOLON = ';';
60  		char SQUARE_BRACKET_CLOSE = ']';
61  		char SQUARE_BRACKET_OPEN = '[';
62  		char WHITESPACE = ' ';
63  	}
64  
65  	interface Tag {
66  		String BROWSER = "browser";
67  		String BROWSER_OS = "browser_os";
68  		String BROWSER_REG = "browser_reg";
69  		String BROWSER_TYPE = "browser_type";
70  		String DEVICE = "device";
71  		String DEVICE_REG = "device_reg";
72  		String OS = "os";
73  		String OS_REG = "os_reg";
74  		String ROBOTS = "robots";
75  	}

77  	private static final String EMPTY = "";

79  	private static void createBrowser(final Browser browser, final StringBuilder builder) {
80  		createKeyValuePair(browser, String.valueOf(browser.getType().getId()), builder);
81  		createKeyValuePair(browser, browser.getFamilyName(), builder);
82  		createKeyValuePair(browser, browser.getUrl(), builder);
83  		createKeyValuePair(browser, browser.getProducer(), builder);
84  		createKeyValuePair(browser, browser.getProducerUrl(), builder);
85  		createKeyValuePair(browser, browser.getIcon(), builder);
86  		createKeyValuePair(browser, browser.getInfoUrl(), builder);
87  	}
88  
89  	private static void createBrowserOperatingSystemMappings(final Data data, final StringBuilder builder) {
90  		final List<BrowserOperatingSystemMapping> mappings = new ArrayList<BrowserOperatingSystemMapping>(
91  				data.getBrowserToOperatingSystemMappings());
92  		Collections.sort(mappings, BrowserOperatingSystemMappingComparator.INSTANCE);
93  		createCategory(Tag.BROWSER_OS, builder);
94  		createComment("browser_id[] = \"OS id\"", builder);
95  		for (final BrowserOperatingSystemMapping mapping : mappings) {
96  			createKeyValuePair(mapping.getBrowserId(), String.valueOf(mapping.getOperatingSystemId()), builder);
97  		}
98  	}

100 	private static void createBrowserPatterns(final Data data, final StringBuilder builder) {
101 		final List<BrowserPattern> patterns = new ArrayList<BrowserPattern>(data.getBrowserPatterns().size());
102 		for (final Entry<Integer, SortedSet<BrowserPattern>> entry : data.getBrowserPatterns().entrySet()) {
103 			patterns.addAll(entry.getValue());
104 		}
105 		Collections.sort(patterns, new OrderedPatternComparator<BrowserPattern>());
106 		createCategory(Tag.BROWSER_REG, builder);
107 		createComment("browser_reg_id[] = \"Browser regstring\"", builder);
108 		createComment("browser_reg_id[] = \"Browser id\"", builder);
109 		for (final BrowserPattern pattern : patterns) {
110 			final String regex = RegularExpressionConverter.convertPatternToPerlRegex(pattern.getPattern());
111 			createKeyValuePair(pattern.getPosition(), regex, builder);
112 			createKeyValuePair(pattern.getPosition(), String.valueOf(pattern.getId()), builder);
113 		}
114 	}

116 	private static void createBrowsers(final Data data, final StringBuilder builder) {
117 		createCategory(Tag.BROWSER, builder);
118 		createComment("browser_id[] = \"Browser type\"", builder);
119 		createComment("browser_id[] = \"Browser Name\"", builder);
120 		createComment("browser_id[] = \"Browser URL\"", builder);
121 		createComment("browser_id[] = \"Browser Company\"", builder);
122 		createComment("browser_id[] = \"Browser Company URL\"", builder);
123 		createComment("browser_id[] = \"Browser ico\"", builder);
124 		createComment("browser_id[] = \"Browser info URL\"", builder);
125 		final List<Browser> browsers = new ArrayList<Browser>(data.getBrowsers());
126 		Collections.sort(browsers, IdentifiableComparator.INSTANCE);
127 		for (final Browser browser : browsers) {
128 			createBrowser(browser, builder);
129 		}
130 	}

132 	private static void createBrowserTypes(final Data data, final StringBuilder builder) {
133 		createCategory(Tag.BROWSER_TYPE, builder);
134 		createComment("browser_type_id[] = \"Browser type\"", builder);
135 		final List<BrowserType> browserTypes = new ArrayList<BrowserType>(data.getBrowserTypes().values());
136 		Collections.sort(browserTypes, IdentifiableComparator.INSTANCE);
137 		for (final BrowserType browserType : browserTypes) {
138 			createKeyValuePair(browserType, browserType.getName(), builder);
139 		}
140 	}

142 	private static void createCategory(@Nonnull final String category, @Nonnull final StringBuilder builder) {
143 		builder.append(Char.SQUARE_BRACKET_OPEN);
144 		builder.append(category);
145 		builder.append(Char.SQUARE_BRACKET_CLOSE);
146 		builder.append(Char.NEWLINE);
147 	}
148 
149 	private static void createComment(@Nonnull final String comment, @Nonnull final StringBuilder builder) {
150 		builder.append(Char.SEMICOLON);
151 		builder.append(Char.WHITESPACE);
152 		builder.append(comment);
153 		builder.append(Char.NEWLINE);
154 	}
155 
156 	private static void createDescription(@Nonnull final Data data, @Nonnull final StringBuilder builder) {
157 		createComment("Data (format ini) for UASparser - http://user-agent-string.info/download/UASparser", builder);
158 		createComment("Version: " + data.getVersion(), builder);
159 		createComment("Checksum:", builder);
160 		createComment("MD5 - http://user-agent-string.info/rpc/get_data.php?format=ini&md5=y", builder);
161 		createComment("SHA1 - http://user-agent-string.info/rpc/get_data.php?format=ini&sha1=y", builder);
162 		builder.append(Char.SEMICOLON);
163 		builder.append(Char.NEWLINE);
164 	}

200 	private static void createKeyValuePair(@Nonnull final Identifiable identifiable, @Nonnull final String value,
201 			@Nonnull final StringBuilder builder) {
202 		createKeyValuePair(identifiable.getId(), value, builder);
203 	}

205 	private static void createKeyValuePair(@Nonnull final int id, @Nonnull final String value, @Nonnull final StringBuilder builder) {
206 		builder.append(id);
207 		builder.append(Char.SQUARE_BRACKET_OPEN);
208 		builder.append(Char.SQUARE_BRACKET_CLOSE);
209 		builder.append(Char.WHITESPACE);
210 		builder.append(Char.EQUALS);
211 		builder.append(Char.WHITESPACE);
212 		builder.append(Char.QUOTE);
213 		builder.append(value);
214 		builder.append(Char.QUOTE);
215 		builder.append(Char.NEWLINE);
216 	}

287 	/**
288 	 * Transforms a given {@code Data} instance into XML and writes it to the passed in {@code OutputStream}.
289 	 * 
290 	 * @param data
291 	 *            {@code Data} to transform into XML
292 	 * @param outputStream
293 	 *            output stream to write
294 	 * @throws IOException
295 	 *             if the given output stream can not be written
296 	 */
297 	public static void write(@Nonnull final Data data, @Nonnull final OutputStream outputStream) throws IOException {
298 		Check.notNull(data, "data");
299 		Check.notNull(outputStream, "outputStream");
300 
301 		final StringBuilder doc = new StringBuilder(10000);
302 
303 		// description element
304 		createDescription(data, doc);
305 
306 		// data
307 		createRobots(data, doc);
308 		createOperatingSystems(data, doc);
309 		createBrowsers(data, doc);
310 		createBrowserTypes(data, doc);
311 		createBrowserPatterns(data, doc);
312 		createBrowserOperatingSystemMappings(data, doc);
313 		createOperatingSystemPatterns(data, doc);
314 		createDevices(data, doc);
315 		createDevicePatterns(data, doc);
316 
317 		// write the content to output stream
318 		outputStream.write(doc.toString().getBytes("UTF-8"));
319 	}
