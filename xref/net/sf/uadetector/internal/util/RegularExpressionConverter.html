33  	public enum Flag {
34  
35  		/**
36  		 * Enables canonical equivalence.
37  		 */
38  		CANON_EQ(Pattern.CANON_EQ, 'c'),
39  
40  		/**
41  		 * Enables case-insensitive matching.
42  		 */
43  		CASE_INSENSITIVE(Pattern.CASE_INSENSITIVE, 'i'),
44  
45  		/**
46  		 * Permits whitespace and comments in pattern.
47  		 */
48  		COMMENTS(Pattern.COMMENTS, 'x'),
49  
50  		/**
51  		 * Enables dotall mode.
52  		 */
53  		DOTALL(Pattern.DOTALL, 's'),
54  
55  		/**
56  		 * Enables literal parsing of the pattern.
57  		 */
58  		LITERAL(Pattern.LITERAL, 'l'),
59  
60  		/**
61  		 * Enables multiline mode.
62  		 */
63  		MULTILINE(Pattern.MULTILINE, 'm'),
64  
65  		/**
66  		 * Enables Unicode-aware case folding.
67  		 */
68  		UNICODE_CASE(Pattern.UNICODE_CASE, 'u'),
69  
70  		/**
71  		 * Enables Unix lines mode.
72  		 */
73  		UNIX_LINES(Pattern.UNIX_LINES, 'e');
74  
75  		private static class FlagByCharacterComparator extends CompareNullSafe<Flag> {
76  			private static final long serialVersionUID = 1L;
77  
78  			@Override
79  			public int compareType(@Nonnull final Flag f1, @Nonnull final Flag f2) {
80  				final Character c1 = Character.valueOf(f1.getCharacter());
81  				final Character c2 = Character.valueOf(f2.getCharacter());
82  				return c1.compareTo(c2);
83  			}
84  		}
85  
86  		private static final FlagByCharacterComparator FLAG_COMPARATOR = new FlagByCharacterComparator();
87  
88  		/**
89  		 * Converts a set of flags as to a bitmask (sum of numerical values).
90  		 * 
91  		 * @param flags
92  		 *            a set of flags
93  		 * @return sum of numerical values of passed flags or 0
94  		 */
95  		public static int convertToBitmask(@Nonnull final Collection<Flag> flags) {
96  			Check.notNull(flags, "flags");
97  
98  			int bitmask = 0;
99  			for (final Flag flag : flags) {
100 				bitmask = bitmask | flag.getNumber();
101 			}
102 			return bitmask;
103 		}
104 
105 		/**
106 		 * Converts a set of flags as to a string representation. The flags {@link Flag#CASE_INSENSITIVE},
107 		 * {@link Flag#DOTALL}, {@link Flag#MULTILINE} and {@link Flag#COMMENTS} are identical to the PERL regular
108 		 * expression modifiers.
109 		 * 
110 		 * @param flags
111 		 *            a set of flags
112 		 * @return sum of numerical values of passed flags or 0
113 		 */
114 		public static String convertToModifiers(@Nonnull final Collection<Flag> flags) {
115 			Check.notNull(flags, "flags");
116 
117 			final StringBuilder modifiers = new StringBuilder(8);
118 			final Set<Flag> sortedFlags = new TreeSet<Flag>(Collections.reverseOrder(FLAG_COMPARATOR));
119 			sortedFlags.addAll(flags);
120 			for (final Flag flag : sortedFlags) {
121 				modifiers.append(flag.getCharacter());
122 			}
123 			return modifiers.toString();
124 		}
125 
126 		/**
127 		 * This method try to find a matching enum value by the given character representation. The character will be
128 		 * evaluated against the stored character of a flag.
129 		 * 
130 		 * @param flag
131 		 *            representation of a flag as a character
132 		 * @return the matching enum value or {@code null}
133 		 * @throws net.sf.qualitycheck.exception.IllegalNegativeArgumentException
134 		 *             if the given number is smaller than zero
135 		 */
136 		public static Flag evaluateByCharacter(final char flag) {
137 			Check.notNegative(flag, "flag");
138 			Flag result = null;
139 			for (final Flag value : values()) {
140 				if (value.getCharacter() == flag) {
141 					result = value;
142 					break;
143 				}
144 			}
145 			return result;
146 		}
147 
148 		/**
149 		 * This method try to find a matching enum value by the given numerical representation. The number will be
150 		 * evaluated against the stored number of a flag.
151 		 * 
152 		 * @param flag
153 		 *            representation of a flag as a character
154 		 * @return the matching enum value or {@code null}
155 		 * @throws net.sf.qualitycheck.exception.IllegalNegativeArgumentException
156 		 *             if the given number is smaller than zero
157 		 */
158 		public static Flag evaluateByNumber(final int flag) {
159 			Check.notNegative(flag, "flag");
160 			Flag result = null;
161 			for (final Flag value : values()) {
162 				if (value.getNumber() == flag) {
163 					result = value;
164 					break;
165 				}
166 			}
167 			return result;
168 		}
169 
170 		/**
171 		 * Parses a sum of flags as numerical values (bitmask) and translates it to set of enum values.
172 		 * 
173 		 * @param bitmask
174 		 *            Sum of numerical values of flags
175 		 * @return a set of flags
176 		 * @throws net.sf.qualitycheck.exception.IllegalNegativeArgumentException
177 		 *             if the given number is smaller than zero
178 		 */
179 		@Nonnull
180 		public static Set<Flag> parse(@Nonnegative final int bitmask) {
181 			Check.notNegative(bitmask, "bitmask");
182 
183 			final Set<Flag> flags = new HashSet<Flag>();
184 			for (final Flag flag : values()) {
185 				if ((bitmask & flag.getNumber()) != 0) {
186 					flags.add(flag);
187 				}
188 			}
189 			return flags;
190 		}
191 
192 		/**
193 		 * Translates PERL style modifiers to a set of {@code Pattern} compatible ones.
194 		 * 
195 		 * @param modifiers
196 		 *            modifiers as string of a PERL style regular expression
197 		 * @return a set of modifier flags that may include CASE_INSENSITIVE, MULTILINE, DOTALL and COMMENTS
198 		 */
199 		public static Set<Flag> parse(@Nonnull final String modifiers) {
200 			Check.notNull(modifiers, "modifiers");
201 
202 			final Set<Flag> flags = new HashSet<Flag>();
203 			for (int i = 0; i < modifiers.length(); i++) {
204 				final Flag flag = Flag.evaluateByCharacter(modifiers.charAt(i));
205 				if (flag != null) {
206 					flags.add(flag);
207 				}
208 			}
209 			return flags;
210 		}
211 
212 		/**
213 		 * Representation of a flag as a character
214 		 */
215 		private final char character;
216 
217 		/**
218 		 * Representation of a flag as a number
219 		 */
220 		private final int number;
221 
222 		private Flag(final int value, final char character) {
223 			number = value;
224 			this.character = character;
225 		}

227 		/**
228 		 * Returns this flag as character representation.
229 		 * 
230 		 * @return representation as a character

234 		}
235 
236 		/**
237 		 * Returns this flag as numerical representation.
238 		 * 
239 		 * @return representation as a number
240 		 */

242 			return number;
243 		}

247 	/**
248 	 * Template to support the conversion into a PERL style regular expression
249 	 */
250 	private static final String PATTERN_TO_REGEX_TEMPLATE = "/%s/%s";
251 
252 	/**
253 	 * Pattern for PERL style regular expression strings
254 	 */
255 	private static final Pattern PERL_STYLE = Pattern.compile("^/.*/((i|m|s|x)*)?$");
256 
257 	/**
258 	 * Pattern for PERL style regular expression strings with more fault-tolerance to the modifiers
259 	 */
260 	private static final Pattern PERL_STYLE_TOLERANT = Pattern.compile("^/.*/(([A-z])*)?$");
261 
262 	/**
263 	 * Converts a given {@code Pattern} into a PERL style regular expression.
264 	 * 
265 	 * @param pattern
266 	 *            regular expression pattern
267 	 * @return PERL style regular expression as string
268 	 */
269 	public static String convertPatternToPerlRegex(@Nonnull final Pattern pattern) {
270 		Check.notNull(pattern, "pattern");
271 		final String modifiers = Flag.convertToModifiers(Flag.parse(pattern.flags()));
272 		return String.format(PATTERN_TO_REGEX_TEMPLATE, pattern.pattern(), modifiers);
273 	}
274 
275 	/**
276 	 * Converts a PERL style regular expression into Java style.<br>
277 	 * <br>
278 	 * The leading and ending slash and the modifiers will be removed. The modifiers will be translated into equivalents
279 	 * flags of <code>java.util.Pattern</code>. If there are modifiers that are not valid an exception will be thrown.
280 	 * 
281 	 * @param regex
282 	 *            A PERL style regular expression
283 	 * @return Pattern
284 	 */
285 	public static Pattern convertPerlRegexToPattern(@Nonnull final String regex) {
286 		return convertPerlRegexToPattern(regex, false);
287 	}
288 
289 	/**
290 	 * Converts a PERL style regular expression into Java style.<br>
291 	 * <br>
292 	 * The leading and ending slash and the modifiers will be removed.
293 	 * 
294 	 * @param regex
295 	 *            A PERL style regular expression
296 	 * @param faultTolerant
297 	 *            Fault-tolerant translating the flags
298 	 * @return Pattern
299 	 */
300 	public static Pattern convertPerlRegexToPattern(@Nonnull final String regex, @Nonnull final boolean faultTolerant) {
301 		Check.notNull(regex, "regex");
302 
303 		String pattern = regex.trim();
304 		final Matcher matcher = faultTolerant ? PERL_STYLE_TOLERANT.matcher(pattern) : PERL_STYLE.matcher(pattern);
305 		if (!matcher.matches()) {
306 			throw new IllegalArgumentException("The given regular expression '" + pattern
307 					+ "' seems to be not in PERL style or has unsupported modifiers.");
308 		}
309 
310 		pattern = pattern.substring(1);
311 		final int lastIndex = pattern.lastIndexOf('/');
312 		pattern = pattern.substring(0, lastIndex);
313 
314 		final int flags = Flag.convertToBitmask(Flag.parse(matcher.group(1)));
315 		return Pattern.compile(pattern, flags);
316 	}
