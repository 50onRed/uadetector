16  package net.sf.uadetector.parser;
17  
18  import java.util.Map.Entry;
19  import java.util.regex.Matcher;
20  
21  import javax.annotation.Nonnull;
22  
23  import net.sf.uadetector.DeviceCategory;
24  import net.sf.uadetector.ReadableDeviceCategory.Category;
25  import net.sf.uadetector.UserAgent;
26  import net.sf.uadetector.UserAgentStringParser;
27  import net.sf.uadetector.UserAgentType;
28  import net.sf.uadetector.VersionNumber;
29  import net.sf.uadetector.datastore.DataStore;
30  import net.sf.uadetector.internal.data.Data;
31  import net.sf.uadetector.internal.data.domain.Browser;
32  import net.sf.uadetector.internal.data.domain.BrowserPattern;
33  import net.sf.uadetector.internal.data.domain.Device;
34  import net.sf.uadetector.internal.data.domain.DevicePattern;
35  import net.sf.uadetector.internal.data.domain.OperatingSystem;
36  import net.sf.uadetector.internal.data.domain.OperatingSystemPattern;
37  import net.sf.uadetector.internal.data.domain.Robot;
38  
39  public abstract class AbstractUserAgentStringParser implements UserAgentStringParser {
40  
41  	/**
42  	 * The number of capturing groups if nothing matches
43  	 */
44  	private static final int ZERO_MATCHING_GROUPS = 0;
45  
46  	/**
47  	 * Examines the user agent string whether it is a browser.
48  	 * 
49  	 * @param userAgent
50  	 *            String of an user agent
51  	 * @param builder
52  	 *            Builder for an user agent information
53  	 */
54  	private static void examineAsBrowser(final UserAgent.Builder builder, final Data data) {
55  		Matcher matcher;
56  		VersionNumber version = VersionNumber.UNKNOWN;
57  		for (final Entry<BrowserPattern, Browser> entry : data.getPatternToBrowserMap().entrySet()) {
58  			matcher = entry.getKey().getPattern().matcher(builder.getUserAgentString());
59  			if (matcher.find()) {
60  
61  				entry.getValue().copyTo(builder);
62  
63  				// try to get the browser version from the first subgroup
64  				if (matcher.groupCount() > ZERO_MATCHING_GROUPS) {
65  					version = VersionNumber.parseVersion(matcher.group(1) != null ? matcher.group(1) : "");
66  				}
67  				builder.setVersionNumber(version);
68  
69  				break;
70  			}
71  		}
72  	}
73  
74  	/**
75  	 * Examines the user agent string whether it is a robot.
76  	 * 
77  	 * @param userAgent
78  	 *            String of an user agent
79  	 * @param builder
80  	 *            Builder for an user agent information
81  	 * @return {@code true} if it is a robot, otherwise {@code false}
82  	 */
83  	private static boolean examineAsRobot(final UserAgent.Builder builder, final Data data) {
84  		boolean isRobot = false;
85  		VersionNumber version;
86  		for (final Robot robot : data.getRobots()) {
87  			if (robot.getUserAgentString().equals(builder.getUserAgentString())) {
88  				isRobot = true;
89  				robot.copyTo(builder);
90  
91  				// try to get the version from the last found group
92  				version = VersionNumber.parseLastVersionNumber(robot.getName());
93  				builder.setVersionNumber(version);
94  
95  				break;
96  			}
97  		}
98  		return isRobot;
99  	}
100 
101 	/**
102 	 * Examines the user agent string whether has a specific device category.
103 	 * 
104 	 * @param userAgent
105 	 *            String of an user agent
106 	 * @param builder
107 	 *            Builder for an user agent information
108 	 */
109 	private static void examineDeviceCategory(final UserAgent.Builder builder, final Data data) {
110 
111 		// a robot will be classified as 'Other'
112 		if (UserAgentType.ROBOT == builder.getType()) {
113 			final DeviceCategory category = findDeviceCategoryByValue(Category.OTHER, data);
114 			builder.setDeviceCategory(category);
115 			return;
116 		}
117 
118 		// classification depends on matching order
119 		for (final Entry<DevicePattern, Device> entry : data.getPatternToDeviceMap().entrySet()) {
120 			final Matcher matcher = entry.getKey().getPattern().matcher(builder.getUserAgentString());
121 			if (matcher.find()) {
122 				final Category category = Category.evaluate(entry.getValue().getName());
123 				final DeviceCategory deviceCategory = findDeviceCategoryByValue(category, data);
124 				builder.setDeviceCategory(deviceCategory);
125 				return;
126 			}
127 		}
128 
129 		// an unknown user agent type should lead to an unknown device
130 		if (UserAgentType.UNKNOWN == builder.getType()) {
131 			builder.setDeviceCategory(DeviceCategory.EMPTY);
132 			return;
133 		}
134 
135 		// if no pattern is available but the type is Other, Library, Validator or UA Anonymizer
136 		// than classify it as 'Other'
137 		if (UserAgentType.OTHER == builder.getType() || UserAgentType.LIBRARY == builder.getType()
138 				|| UserAgentType.VALIDATOR == builder.getType() || UserAgentType.USERAGENT_ANONYMIZER == builder.getType()) {
139 			final DeviceCategory category = findDeviceCategoryByValue(Category.OTHER, data);
140 			builder.setDeviceCategory(category);
141 			return;
142 		}
143 
144 		// if no pattern is available but the type is a mobile or WAP browser than classify it as 'Smartphone'
145 		if (UserAgentType.MOBILE_BROWSER == builder.getType() || UserAgentType.WAP_BROWSER == builder.getType()) {
146 			final DeviceCategory category = findDeviceCategoryByValue(Category.SMARTPHONE, data);
147 			builder.setDeviceCategory(category);
148 			return;
149 		}
150 
151 		final DeviceCategory category = findDeviceCategoryByValue(Category.PERSONAL_COMPUTER, data);
152 		builder.setDeviceCategory(category);
153 	}
154 
155 	/**
156 	 * Examines the operating system of the user agent string, if not available.
157 	 * 
158 	 * @param userAgent
159 	 *            String of an user agent
160 	 * @param builder
161 	 *            Builder for an user agent information
162 	 */
163 	private static void examineOperatingSystem(final UserAgent.Builder builder, final Data data) {
164 		if (net.sf.uadetector.OperatingSystem.EMPTY.equals(builder.getOperatingSystem())) {
165 			for (final Entry<OperatingSystemPattern, OperatingSystem> entry : data.getPatternToOperatingSystemMap().entrySet()) {
166 				final Matcher matcher = entry.getKey().getPattern().matcher(builder.getUserAgentString());
167 				if (matcher.find()) {
168 					entry.getValue().copyTo(builder);
169 					break;
170 				}
171 			}
172 		}
173 	}
174 
175 	private static DeviceCategory findDeviceCategoryByValue(@Nonnull final Category category, @Nonnull final Data data) {
176 		for (final Device device : data.getDevices()) {
177 			if (category == device.getCategory()) {
178 				return new DeviceCategory(category, device.getIcon(), device.getInfoUrl(), device.getName());
179 			}
180 		}
181 		return DeviceCategory.EMPTY;
182 	}
183 
184 	/**
185 	 * Gets the data store of this parser.
186 	 * 
187 	 * @return data store of this parser
188 	 */
189 	protected abstract DataStore getDataStore();
190 
191 	@Override
192 	public String getDataVersion() {
193 		return getDataStore().getData().getVersion();
194 	}
195 
196 	@Override
197 	public UserAgent parse(final String userAgent) {
198 		final UserAgent.Builder builder = new UserAgent.Builder(userAgent);
199 
200 		// work during the analysis always with the same reference of data
201 		final Data data = getDataStore().getData();
202 
203 		if (!examineAsRobot(builder, data)) {
204 			examineAsBrowser(builder, data);
205 			examineOperatingSystem(builder, data);
206 		}
207 		examineDeviceCategory(builder, data);
208 		return builder.build();
209 	}
210 
211 	@Override
212 	public void shutdown() {
213 		// nothing to shutdown
214 	}
215
