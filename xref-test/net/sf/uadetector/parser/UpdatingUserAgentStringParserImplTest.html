366 	@Test
367 	public void testUpdateMechanismWhileParsing() throws InterruptedException {
368 		final UpdatingUserAgentStringParserImpl parser = new UpdatingUserAgentStringParserImpl(new TestXmlDataStore());
369 		final long firstLastUpdateCheck = parser.getDataStore().getUpdateOperation().getLastUpdateCheck();
370 		LOG.debug("LastUpdateCheck at: " + firstLastUpdateCheck);
371 		final long originalInterval = parser.getUpdateInterval();
372 
373 		// reduce the interval since testing
374 		LOG.debug("Reducing the update interval during the test.");
375 		parser.setUpdateInterval(400l);
376 		// we have to read to activate the update mechanism
377 		parser.parse("check for updates");
378 
379 		Thread.sleep(1000l);
380 		final long currentLastUpdateCheck = parser.getDataStore().getUpdateOperation().getLastUpdateCheck();
381 		LOG.debug("LastUpdateCheck at: " + currentLastUpdateCheck);
382 		assertThat(firstLastUpdateCheck < currentLastUpdateCheck).isTrue();
383 
384 		parser.setUpdateInterval(originalInterval);
385 	}
