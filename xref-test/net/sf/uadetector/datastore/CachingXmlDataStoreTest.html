122 	@Test
123 	public void createCachingXmlDataStore_provokeFallback() throws IOException, InterruptedException {
124 		// create temp file
125 		final File cache = folder.newFile("uas_temp.xml");
126 		assertThat(readFile(cache)).isEqualTo("");
127 
128 		// create fallback data store
129 		final String version = "fallback-data-version";
130 		final DataStore fallback = new DataStore() {
131 
132 			@Override
133 			public Charset getCharset() {
134 				return null;
135 			}
136 
137 			@Override
138 			public Data getData() {
139 				return new DataBlueprint().version(version).build();
140 			}
141 
142 			@Override
143 			public DataReader getDataReader() {
144 				return null;
145 			}
146 
147 			@Override
148 			public URL getDataUrl() {
149 				return null;
150 			}
151 
152 			@Override
153 			public URL getVersionUrl() {
154 				return null;
155 			}
156 		};
157 
158 		// create caching data store
159 		final DataStore store = CachingXmlDataStore.createCachingXmlDataStore(cache, UNREACHABLE_URL, UNREACHABLE_URL, CHARSET, fallback);
160 		assertThat(store.getData().getVersion()).isEqualTo(version);
161 	}

163 	@Test
164 	public void createCachingXmlDataStore_successful() throws IOException, InterruptedException {
165 		// create temp file
166 		final File cache = folder.newFile("uas_temp.xml");
167 		assertThat(readFile(cache)).isEqualTo("");
168 
169 		// create caching data store
170 		final CachingXmlDataStore store = CachingXmlDataStore.createCachingXmlDataStore(cache, DATA_URL, VERSION_URL, CHARSET, fallback);
171 		final UpdatingUserAgentStringParserImpl parser = new UpdatingUserAgentStringParserImpl(store);
172 
173 		Thread.sleep(1000l);
174 		assertThat(readFile(cache).length() >= 721915).isTrue();
175 
176 		final long firstLastUpdateCheck = store.getUpdateOperation().getLastUpdateCheck();
177 		LOG.debug("LastUpdateCheck at: " + firstLastUpdateCheck);
178 		final long originalInterval = parser.getUpdateInterval();
179 
180 		// reduce the interval since testing
181 		LOG.debug("Reducing the update interval during the test.");
182 		parser.setUpdateInterval(1000l);
183 		// we have to read to activate the update mechanism
184 		parser.parse("check for updates");
185 
186 		Thread.sleep(3000l);
187 		final long currentLastUpdateCheck = store.getUpdateOperation().getLastUpdateCheck();
188 		LOG.debug("LastUpdateCheck at: " + currentLastUpdateCheck);
189 		assertThat(firstLastUpdateCheck < currentLastUpdateCheck).isTrue();
190 
191 		parser.setUpdateInterval(originalInterval);
192 	}

199 	@Test
200 	public void findOrCreateCacheFile() {
201 		CachingXmlDataStore.findOrCreateCacheFile().delete(); // delete if exists
202 		File temp = CachingXmlDataStore.findOrCreateCacheFile(); // cache file does not exist
203 		assertThat(temp.exists()).isTrue();
204 		temp = CachingXmlDataStore.findOrCreateCacheFile(); // cache file exists
205 		assertThat(temp.exists()).isTrue();
206 		temp = CachingXmlDataStore.findOrCreateCacheFile(); // cache file exists
207 		assertThat(temp.exists()).isTrue();
208 		temp.delete();
209 		assertThat(temp.exists()).isFalse();
210 	}
