3   import static org.fest.assertions.Assertions.assertThat;
4   import org.junit.Test;
5   
6   public class AbstractUpdateOperationTest {
7   
8   	@Test
9   	public void hasUpdate_true() {
10  		assertThat(AbstractUpdateOperation.hasUpdate("20130310-02", "20130310-01")).isTrue();
11  		assertThat(AbstractUpdateOperation.hasUpdate("20130310-01", "20130301-01")).isTrue();
12  		assertThat(AbstractUpdateOperation.hasUpdate("20130310-02", "20120408-10")).isTrue();
13  	}
14  
15  	@Test
16  	public void hasUpdate_false() {
17  		assertThat(AbstractUpdateOperation.hasUpdate("20130310-02", "20130310-03")).isFalse();
18  		assertThat(AbstractUpdateOperation.hasUpdate("20130310-01", "20130311-01")).isFalse();
19  		assertThat(AbstractUpdateOperation.hasUpdate("20110310-02", "20120408-10")).isFalse();
20  		assertThat(AbstractUpdateOperation.hasUpdate("", "20120408-10")).isFalse();
21  		assertThat(AbstractUpdateOperation.hasUpdate("<h1>Error - Connect failed:<br /> <u>Too many connections</u></h1>", "20120408-10"))
22  				.isFalse();
23  		assertThat(AbstractUpdateOperation.hasUpdate("", "")).isFalse();
24  		assertThat(AbstractUpdateOperation.hasUpdate("", " ")).isFalse();
25  		assertThat(AbstractUpdateOperation.hasUpdate(" ", "")).isFalse();
26  		assertThat(AbstractUpdateOperation.hasUpdate("20131011-02", "unknown-pattern")).isFalse();
27  	}
28  
29  }
