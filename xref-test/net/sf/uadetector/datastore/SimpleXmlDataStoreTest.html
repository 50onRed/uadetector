16  package net.sf.uadetector.datastore;
17  
18  import java.net.URL;
19  
20  import net.sf.uadetector.internal.util.UrlUtil;
21  
22  import static org.fest.assertions.Assertions.assertThat;
23  import org.junit.Test;
24  
25  public class SimpleXmlDataStoreTest {
26  
27  	/**
28  	 * URL to retrieve the UAS data as XML
29  	 */
30  	private static final URL DATA_URL = AbstractDataStoreTest2.class.getClassLoader().getResource("uas_newer.xml");
31  
32  	/**
33  	 * {@link URL} that is not available
34  	 */
35  	private static final URL UNREACHABLE_URL = UrlUtil.build("http://unreachable.local/");
36  
37  	/**
38  	 * URL to retrieve the version of the UAS data
39  	 */
40  	private static final URL VERSION_URL = AbstractDataStoreTest2.class.getClassLoader().getResource("uas_newer.version");
41  
42  	@Test
43  	public void construct_successful() {
44  		final SimpleXmlDataStore store = new SimpleXmlDataStore(DATA_URL, VERSION_URL);
45  		assertThat(!store.getData().getVersion().isEmpty()).isTrue();
46  		assertThat(store.getDataReader()).isNotNull();
47  		assertThat(store.getDataUrl()).isNotNull();
48  		assertThat(store.getVersionUrl()).isNotNull();
49  	}
50  
51  	@Test(expected = IllegalStateException.class)
52  	public void readData_fails() {
53  		new SimpleXmlDataStore(UNREACHABLE_URL, UNREACHABLE_URL);
54  	}
55  
56  }
