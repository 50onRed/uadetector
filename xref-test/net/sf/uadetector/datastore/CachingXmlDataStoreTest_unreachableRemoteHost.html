16  package net.sf.uadetector.datastore;
17  
18  import static org.fest.assertions.Assertions.assertThat;
19  
20  import java.io.File;
21  import java.io.IOException;
22  import java.net.URL;
23  import java.nio.charset.Charset;
24  
25  import net.sf.uadetector.internal.util.UrlUtil;
26  import net.sf.uadetector.parser.UpdatingUserAgentStringParserImpl;
27  
28  import org.junit.Rule;
29  import org.junit.Test;
30  import org.junit.rules.TemporaryFolder;
31  
32  import com.google.common.io.Files;
33  
34  public class CachingXmlDataStoreTest_unreachableRemoteHost {
35  
36  	/**
37  	 * The character set to read UAS data
38  	 */
39  	private static final Charset CHARSET = DataStore.DEFAULT_CHARSET;
40  
41  	/**
42  	 * Time to wait before the test can be aborted (in milliseconds)
43  	 */
44  	private static final long WAIT_UNTIL_ABORT = 30000;
45  
46  	/**
47  	 * URL to retrieve a newer UAS data as XML
48  	 */
49  	private static final URL DATA_URL = TestXmlDataStore.class.getClassLoader().getResource("uas_newer.xml");
50  
51  	/**
52  	 * Version of the newer UAS data
53  	 */
54  	private static final String DATA_VERSION = TestXmlDataStore.VERSION_NEWER;
55  
56  	/**
57  	 * {@link URL} that is not available
58  	 */
59  	private static final URL UNREACHABLE_URL = UrlUtil.build("http://unreachable.local/");
60  
61  	/**
62  	 * URL to retrieve a newer UAS data as XML
63  	 */
64  	private static final URL VERSION_URL = TestXmlDataStore.class.getClassLoader().getResource("uas_newer.version");
65  
66  	private static String readFile(final File file) throws IOException {
67  		byte[] bytes = Files.toByteArray(file);
68  		return new String(bytes, Charset.defaultCharset());
69  	}
70  
71  	/**
72  	 * Temporary folder to cache <em>UAS data</em> in a file. Created files in this folder are guaranteed to be deleted
73  	 * when the test method finishes (whether it passes or fails).
74  	 */
75  	@Rule
76  	public final TemporaryFolder folder = new TemporaryFolder();
77  
78  	@Test
79  	public void loadingOfRemoteData_successfully() throws IOException, InterruptedException {
80  		// setup a temporary file
81  		final File cache = folder.newFile();
82  
83  		// create fallback data store
84  		final TestXmlDataStore fallback = new TestXmlDataStore();
85  
86  		// create caching data store without a cache file
87  		final long startTime = System.currentTimeMillis();
88  		final CachingXmlDataStore store = CachingXmlDataStore.createCachingXmlDataStore(cache, DATA_URL, VERSION_URL, CHARSET, fallback);
89  		assertThat(store.getData().getVersion()).isEqualTo(fallback.getData().getVersion());
90  		final long duration = System.currentTimeMillis() - startTime;
91  		assertThat(duration < 1000).as("loading unreachable remote data takes too long").isTrue();
92  
93  		final UpdatingUserAgentStringParserImpl parser = new UpdatingUserAgentStringParserImpl(store);
94  		parser.parse("test");
95  
96  		final long lookupStartTime = System.currentTimeMillis();
97  		do {
98  			Thread.sleep(1000);
99  		} while (!readFile(cache).contains(DATA_VERSION) && !(System.currentTimeMillis() - lookupStartTime > WAIT_UNTIL_ABORT));
100 		assertThat(store.getData().getVersion()).isEqualTo(DATA_VERSION);
101 
102 		final String readIn = readFile(cache);
103 		assertThat(readIn.contains(DATA_VERSION)).isTrue();
104 		assertThat(readIn.length() >= 721915).isTrue();
105 	}
106 
107 	@Test
108 	public void loadingOfUnreachableRemoteData_checkToBeFast() throws IOException, InterruptedException {
109 		// setup a temporary file
110 		final File cache = folder.newFile();
111 
112 		// create fallback data store
113 		final TestXmlDataStore fallback = new TestXmlDataStore();
114 
115 		// create caching data store without a cache file
116 		final long startTime = System.currentTimeMillis();
117 		final CachingXmlDataStore store = CachingXmlDataStore.createCachingXmlDataStore(cache, UNREACHABLE_URL, UNREACHABLE_URL, CHARSET,
118 				fallback);
119 		assertThat(store.getData().getVersion()).isEqualTo(fallback.getData().getVersion());
120 		final long duration = System.currentTimeMillis() - startTime;
121 		assertThat(duration < 1000).as("loading unreachable remote data takes too long").isTrue();
122 	}
123 
124 	@Test
125 	public void loadingOfUnreachableRemoteData_useFallback() throws IOException, InterruptedException {
126 		// setup a temporary file
127 		final File cache = folder.newFile();
128 
129 		// create fallback data store
130 		final TestXmlDataStore fallback = new TestXmlDataStore();
131 
132 		// create caching data store without a cache file
133 		final CachingXmlDataStore store = CachingXmlDataStore.createCachingXmlDataStore(cache, UNREACHABLE_URL, UNREACHABLE_URL, CHARSET,
134 				fallback);
135 		assertThat(store.getData().getVersion()).isEqualTo(fallback.getData().getVersion());
136 
137 		final UpdatingUserAgentStringParserImpl parser = new UpdatingUserAgentStringParserImpl(store);
138 		parser.parse("test");
139 
140 		final String newerVersionOfFallback = UrlUtil.read(TestXmlDataStore.VERSION_URL_NEWER, DataStore.DEFAULT_CHARSET);
141 		final long lookupStartTime = System.currentTimeMillis();
142 		do {
143 			Thread.sleep(1000l);
144 		} while (!readFile(cache).contains(newerVersionOfFallback) && !(System.currentTimeMillis() - lookupStartTime > WAIT_UNTIL_ABORT));
145 
146 		assertThat(store.getData().getVersion()).isEqualTo(newerVersionOfFallback);
147 		assertThat(readFile(cache).contains(newerVersionOfFallback)).isTrue();
148 	}
149 
150 }
