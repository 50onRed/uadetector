16  package net.sf.uadetector.internal.data.domain;
17  
18  import net.sf.qualitycheck.exception.IllegalNegativeArgumentException;
19  import net.sf.qualitycheck.exception.IllegalNullArgumentException;
20  import net.sf.uadetector.internal.data.domain.BrowserType.Builder;
21  
22  import static org.fest.assertions.Assertions.assertThat;
23  import org.junit.Test;
24  
25  public class BrowserTypeBuilderTest {
26  
27  	@Test(expected = IllegalNegativeArgumentException.class)
28  	public void setId_toSmall() {
29  		Builder b = new BrowserType.Builder();
30  		b.setId(-1);
31  	}
32  
33  	@Test
34  	public void setId_byString_valid() {
35  		Builder b = new BrowserType.Builder();
36  		b.setId("1");
37  	}
38  
39  	@Test(expected = NumberFormatException.class)
40  	public void setId_byString_invalid() {
41  		Builder b = new BrowserType.Builder();
42  		b.setId("abc");
43  	}
44  
45  	@Test(expected = IllegalNegativeArgumentException.class)
46  	public void setId_byString_toSmall() {
47  		Builder b = new BrowserType.Builder();
48  		b.setId("-1");
49  	}
50  
51  	@Test(expected = IllegalNullArgumentException.class)
52  	public void setId_null() {
53  		Builder b = new BrowserType.Builder();
54  		b.setId(null);
55  	}
56  
57  	@Test(expected = IllegalNullArgumentException.class)
58  	public void setName_null() {
59  		Builder b = new BrowserType.Builder();
60  		b.setName(null);
61  	}
62  
63  	@Test
64  	public void testSettersAndGetters() {
65  		Builder b = new BrowserType.Builder();
66  		b.setId(1);
67  		b.setName("Test");
68  		BrowserType i = b.build();
69  		assertThat(i.getId()).isEqualTo(1);
70  		assertThat(i.getName()).isEqualTo("Test");
71  	}
72  
73  }
