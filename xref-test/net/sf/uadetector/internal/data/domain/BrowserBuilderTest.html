37  	@Test
38  	public void construct_copy_successful() {
39  		final BrowserType browserType = new BrowserType.Builder().setId("1").setName("browser type test").build();
40  		final SortedSet<OperatingSystemPattern> osPatternSet = new TreeSet<OperatingSystemPattern>();
41  		final OperatingSystem operatingSystem = new OperatingSystem(1, "n1", "f1", "iu1", osPatternSet, "p1", "pu1", "u1", "i1");
42  		final SortedSet<BrowserPattern> patternSet = new TreeSet<BrowserPattern>();
43  
44  		final Browser.Builder original = new Browser.Builder();
45  		original.setFamilyName(UserAgentFamily.CHROME.getName());
46  		original.setIcon("i1");
47  		original.setId(1);
48  		original.setInfoUrl("iu1");
49  		original.setOperatingSystem(operatingSystem);
50  		original.setPatterns(patternSet);
51  		original.setProducer("p1");
52  		original.setProducerUrl("pu1");
53  		original.setType(browserType);
54  		original.setTypeId(1);
55  		original.setUrl("u1");
56  
57  		final Builder copy1 = new Browser.Builder(original);
58  		assertThat(copy1.getFamily()).isEqualTo(UserAgentFamily.CHROME);
59  		assertThat(copy1.getIcon()).isEqualTo("i1");
60  		assertThat(copy1.getInfoUrl()).isEqualTo("iu1");
61  		assertThat(copy1.getOperatingSystem()).isSameAs(operatingSystem);
62  		assertThat(copy1.getPatterns()).isEqualTo(patternSet);
63  		assertThat(copy1.getProducer()).isEqualTo("p1");
64  		assertThat(copy1.getProducerUrl()).isEqualTo("pu1");
65  		assertThat(copy1.getType().getName()).isEqualTo("browser type test");
66  		assertThat(copy1.getType().getId()).isEqualTo(1);
67  		assertThat(copy1.getTypeId()).isEqualTo(1);
68  		assertThat(copy1.getUrl()).isEqualTo("u1");
69  
70  		final Builder copy2 = original.copy();
71  		assertThat(copy2.getFamily()).isEqualTo(UserAgentFamily.CHROME);
72  		assertThat(copy2.getIcon()).isEqualTo("i1");
73  		assertThat(copy2.getInfoUrl()).isEqualTo("iu1");
74  		assertThat(copy2.getOperatingSystem()).isSameAs(operatingSystem);
75  		assertThat(copy2.getPatterns()).isEqualTo(patternSet);
76  		assertThat(copy2.getProducer()).isEqualTo("p1");
77  		assertThat(copy2.getProducerUrl()).isEqualTo("pu1");
78  		assertThat(copy2.getType().getName()).isEqualTo("browser type test");
79  		assertThat(copy2.getType().getId()).isEqualTo(1);
80  		assertThat(copy2.getTypeId()).isEqualTo(1);
81  		assertThat(copy2.getUrl()).isEqualTo("u1");
82  	}

193 	@Test
194 	public void testGetters() throws Exception {
195 		final BrowserType browserType = new BrowserType.Builder().setId("1").setName("browser type test").build();
196 		final SortedSet<OperatingSystemPattern> osPatternSet = new TreeSet<OperatingSystemPattern>();
197 		final OperatingSystem operatingSystem = new OperatingSystem(1, "n1", "f1", "iu1", osPatternSet, "p1", "pu1", "u1", "i1");
198 		final SortedSet<BrowserPattern> patterns = new TreeSet<BrowserPattern>();
199 
200 		final Browser.Builder builder = new Browser.Builder();
201 		assertThat(builder.setFamilyName(UserAgentFamily.SEAMONKEY.getName())).isSameAs(builder);
202 		assertThat(builder.setIcon("i1")).isSameAs(builder);
203 		assertThat(builder.setId(1)).isSameAs(builder);
204 		assertThat(builder.setInfoUrl("iu1")).isSameAs(builder);
205 		assertThat(builder.setOperatingSystem(operatingSystem)).isSameAs(builder);
206 		assertThat(builder.setPatterns(patterns)).isSameAs(builder);
207 		assertThat(builder.setProducer("p1")).isSameAs(builder);
208 		assertThat(builder.setProducerUrl("pu1")).isSameAs(builder);
209 		assertThat(builder.setType(browserType)).isSameAs(builder);
210 		assertThat(builder.setTypeId(1)).isSameAs(builder);
211 		assertThat(builder.setUrl("u1")).isSameAs(builder);
212 
213 		assertThat(builder.getFamily()).isEqualTo(UserAgentFamily.SEAMONKEY);
214 		assertThat(builder.getFamilyName()).isEqualTo(UserAgentFamily.SEAMONKEY.getName());
215 		assertThat(builder.getIcon()).isEqualTo("i1");
216 		assertThat(builder.getInfoUrl()).isEqualTo("iu1");
217 		assertThat(builder.getOperatingSystem()).isSameAs(operatingSystem);
218 		assertThat(builder.getPatterns()).isEqualTo(patterns);
219 		assertThat(builder.getProducer()).isEqualTo("p1");
220 		assertThat(builder.getProducerUrl()).isEqualTo("pu1");
221 		assertThat(builder.getType().getName()).isEqualTo("browser type test");
222 		assertThat(builder.getType().getId()).isEqualTo(1);
223 		assertThat(builder.getTypeId()).isEqualTo(1);
224 		assertThat(builder.getUrl()).isEqualTo("u1");
225 
226 		final Browser browser = builder.build();
227 		assertThat(browser.getFamily()).isEqualTo(UserAgentFamily.SEAMONKEY);
228 		assertThat(browser.getFamilyName()).isEqualTo(UserAgentFamily.SEAMONKEY.getName());
229 		assertThat(browser.getIcon()).isEqualTo("i1");
230 		assertThat(browser.getInfoUrl()).isEqualTo("iu1");
231 		assertThat(browser.getProducer()).isEqualTo("p1");
232 		assertThat(browser.getProducerUrl()).isEqualTo("pu1");
233 		assertThat(browser.getType().getName()).isEqualTo("browser type test");
234 		assertThat(browser.getType().getId()).isEqualTo(1);
235 		assertThat(browser.getUrl()).isEqualTo("u1");
236 
237 		final Browser browserRebuild = new Builder(builder.build()).build();
238 		assertThat(browserRebuild.getFamily()).isEqualTo(UserAgentFamily.SEAMONKEY);
239 		assertThat(browserRebuild.getFamilyName()).isEqualTo(UserAgentFamily.SEAMONKEY.getName());
240 		assertThat(browserRebuild.getIcon()).isEqualTo("i1");
241 		assertThat(browserRebuild.getInfoUrl()).isEqualTo("iu1");
242 		assertThat(browserRebuild.getProducer()).isEqualTo("p1");
243 		assertThat(browserRebuild.getProducerUrl()).isEqualTo("pu1");
244 		assertThat(browserRebuild.getType().getName()).isEqualTo("browser type test");
245 		assertThat(browserRebuild.getType().getId()).isEqualTo(1);
246 		assertThat(browserRebuild.getUrl()).isEqualTo("u1");
247 	}
