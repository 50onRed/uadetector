58  	@Test
59  	public void equals_different_browserPatterns() {
60  		final Browser browser = createBrowser(1);
61  		final BrowserPattern pattern1 = new BrowserPattern(1, Pattern.compile("[0-9]+"), 1);
62  		final BrowserPattern pattern2 = new BrowserPattern(2, Pattern.compile("[0-9]+"), 1);
63  		final BrowserPattern pattern3 = new BrowserPattern(3, Pattern.compile("[0-9]+"), 3);
64  		final BrowserPattern pattern4 = new BrowserPattern(3, Pattern.compile("[0-9]+"), 3);
65  		final TreeMap<BrowserPattern, Browser> patterns1 = new TreeMap<BrowserPattern, Browser>();
66  		patterns1.put(pattern1, browser);
67  		patterns1.put(pattern2, browser);
68  		final TreeMap<BrowserPattern, Browser> patterns2 = new TreeMap<BrowserPattern, Browser>();
69  		patterns2.put(pattern1, browser);
70  		final TreeMap<BrowserPattern, Browser> patterns3 = new TreeMap<BrowserPattern, Browser>();
71  		patterns3.put(pattern3, browser);
72  		final TreeMap<BrowserPattern, Browser> patterns4 = new TreeMap<BrowserPattern, Browser>();
73  		patterns4.put(pattern4, browser);
74  
75  		final Data data1 = new DataBlueprint().patternToBrowserMap(patterns1).build();
76  		final Data data2 = new DataBlueprint().patternToBrowserMap(patterns2).build();
77  		final Data data3 = new DataBlueprint().patternToBrowserMap(patterns3).build();
78  		final Data data4 = new DataBlueprint().patternToBrowserMap(patterns4).build();
79  
80  		assertThat(data1.equals(data2)).isFalse();
81  		assertThat(data1.hashCode() == data2.hashCode()).isFalse();
82  		assertThat(data2.equals(data1)).isFalse();
83  
84  		assertThat(data1.equals(data3)).isFalse();
85  		assertThat(data1.hashCode() == data3.hashCode()).isFalse();
86  		assertThat(data3.equals(data1)).isFalse();
87  
88  		assertThat(data3.equals(data4)).isTrue();
89  		assertThat(data3.hashCode() == data4.hashCode()).isTrue();
90  		assertThat(data4.equals(data3)).isTrue();
91  
92  		assertThat(data2.equals(data3)).isFalse();
93  		assertThat(data2.hashCode() == data3.hashCode()).isFalse();
94  
95  		assertThat(data1.equals(data4)).isFalse();
96  		assertThat(data1.hashCode() == data4.hashCode()).isFalse();
97  	}

117 	public void equals_different_operatingSystemPatterns() {
118 		final OperatingSystem operatingSystem = createOperatingSystem(1);
119 		final OperatingSystemPattern pattern1 = new OperatingSystemPattern(1, Pattern.compile("[0-9]+"), 1);
120 		final OperatingSystemPattern pattern2 = new OperatingSystemPattern(2, Pattern.compile("[0-9]+"), 1);
121 		final OperatingSystemPattern pattern3 = new OperatingSystemPattern(3, Pattern.compile("[0-9]+"), 3);
122 		final OperatingSystemPattern pattern4 = new OperatingSystemPattern(3, Pattern.compile("[0-9]+"), 3);
123 		final TreeMap<OperatingSystemPattern, OperatingSystem> patterns1 = new TreeMap<OperatingSystemPattern, OperatingSystem>();
124 		patterns1.put(pattern1, operatingSystem);
125 		patterns1.put(pattern2, operatingSystem);
126 		final TreeMap<OperatingSystemPattern, OperatingSystem> patterns2 = new TreeMap<OperatingSystemPattern, OperatingSystem>();
127 		patterns2.put(pattern1, operatingSystem);
128 		final TreeMap<OperatingSystemPattern, OperatingSystem> patterns3 = new TreeMap<OperatingSystemPattern, OperatingSystem>();
129 		patterns3.put(pattern3, operatingSystem);
130 		final TreeMap<OperatingSystemPattern, OperatingSystem> patterns4 = new TreeMap<OperatingSystemPattern, OperatingSystem>();
131 		patterns4.put(pattern4, operatingSystem);
132 
133 		final Data data1 = new DataBlueprint().patternToOperatingSystemMap(patterns1).build();
134 		final Data data2 = new DataBlueprint().patternToOperatingSystemMap(patterns2).build();
135 		final Data data3 = new DataBlueprint().patternToOperatingSystemMap(patterns3).build();
136 		final Data data4 = new DataBlueprint().patternToOperatingSystemMap(patterns4).build();
137 
138 		assertThat(data1.equals(data2)).isFalse();
139 		assertThat(data1.hashCode() == data2.hashCode()).isFalse();
140 		assertThat(data2.equals(data1)).isFalse();
141 
142 		assertThat(data1.equals(data3)).isFalse();
143 		assertThat(data1.hashCode() == data3.hashCode()).isFalse();
144 		assertThat(data3.equals(data1)).isFalse();
145 
146 		assertThat(data3.equals(data4)).isTrue();
147 		assertThat(data3.hashCode() == data4.hashCode()).isTrue();
148 		assertThat(data4.equals(data3)).isTrue();
149 
150 		assertThat(data2.equals(data3)).isFalse();
151 		assertThat(data2.hashCode() == data3.hashCode()).isFalse();
152 
153 		assertThat(data1.equals(data4)).isFalse();
154 		assertThat(data1.hashCode() == data4.hashCode()).isFalse();
155 	}
