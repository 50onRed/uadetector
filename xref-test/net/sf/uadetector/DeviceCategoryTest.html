56  	@Test
57  	public void blueprintEqualsBuilder() {
58  		final DeviceCategory blueprint = new Blueprint().category(Category.SMARTPHONE).icon("icn").infoUrl("url").name("a name").build();
59  		final DeviceCategory.Builder builder = new DeviceCategory.Builder();
60  
61  		builder.setCategory(Category.SMARTPHONE);
62  		builder.setIcon("icn");
63  		builder.setInfoUrl("url");
64  		builder.setName("a name");
65  
66  		final DeviceCategory obj = builder.build();
67  		assertThat(obj).isEqualTo(blueprint);
68  	}
